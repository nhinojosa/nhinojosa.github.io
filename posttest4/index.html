If no text is entered and/or the user is not 21 or over you should display an error message (this can be a popup or a message written in the div provided) and do not submit the form to the action page.

        <form name="myFrm" onsubmit="return validateForm();" action="somepg.html">

		First Name: <input type="text" name="firstName"	id="firstName"	/>

        <script>
            function validateForm()
            {
                var fName = document.forms["myFrm"]["firstName"].value;
                var lName = document.forms["myFrm"]["lastName"].value;
                var age = document.forms["myFrm"]["age"].value;

                if( fName == "" || lName == "" || age == "" || age < 21)
                {
                    document.getElementById("errorMsg").innerHTML = "Invalid Data: User must be 21 or older";
                    return false;
                }
                
            }
        </script>

Create a function that will take in 3 arguments. 2 of the arguments should be numeric and the other should be a math operator (+, -, *, /, or %) based on the input you should return the answer after the operation is performed on the 2 integers. 


    <button onclick="calc(4,5,'*');"> Click Me </button>
    <script>
        function calc(x, y, op)
        {
            var num1 = parseInt(x);
            var num2 = parseInt(y);

            if(op == "+")
            {
                document.write(num1 + num2);
            }
            else if (op == "-")
            {
                document.write(num1 - num2);
            }
            else if (op == "*")
            {
                document.write(num1 * num2);
            } 
            else if (op == "/")
            {
                document.write(num1 / num2);
            }           
        }
    </script>
